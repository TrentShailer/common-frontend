/// Serialize some bytes as base 64.
pub fn serialize<S, V: AsRef<[u8]>>(value: &V, serializer: S) -> Result<S::Ok, S::Error>
where
    S: Serializer,
{
    serializer.serialize_str(&Base64UrlUnpadded::encode_string(value.as_ref()))
}
